public class ChessBoard { public static void main(String[] args) { char[][] board = initializeBoard(); printBoard(board); } private static char[][] initializeBoard() { char[][] board = new char[8][8]; // Initialize pieces on the board // 'R' represents rook, 'N' represents knight, 'B' represents bishop // 'Q' represents queen, 'K' represents king, 'P' represents pawn // White pieces board[0] = new char[]{'R', 'N', 'B', 'Q', 'K', 'B', 'N', 'R'}; board[1] = new char[]{'P', 'P', 'P', 'P', 'P', 'P', 'P', 'P'}; // Empty rows for (int i = 2; i < 6; i++) { board[i] = new char[]{' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '}; } // Black pieces board[6] = new char[]{'p', 'p', 'p', 'p', 'p', 'p', 'p', 'p'}; board[7] = new char[]{'r', 'n', 'b', 'q', 'k', 'b', 'n', 'r'}; return board; } private static void printBoard(char[][] board) { for (int i = 0; i < 8; i++) { for (int j = 0; j < 8; j++) { System.out.print(board[i][j] + " "); } System.out.println(); } } }
